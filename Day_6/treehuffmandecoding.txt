void decode(String s, Node root) {
        Node temp=root;
        char[] str=s.toCharArray();  
        for (int i=0;i<s.length();i++) 
        { 
            if (str[i]=='0') 
                temp = temp.left; 
            else
                temp = temp.right;  
        if (temp.left==null && temp.right==null) 
        { 
            System.out.print(temp.data); 
            temp = root; 
        } 
        }
            
    }
